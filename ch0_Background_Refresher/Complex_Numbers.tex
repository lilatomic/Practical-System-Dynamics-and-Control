\chapter{Complex Numbers}

\begin{section}{Representations}
	
	\subsection{Coordinate}
		\begin{equation*}
			z=x+\hat{j}y
		\end{equation*}
	
	\subsection{Phasor}
		\begin{equation*}
			z=Ae^{\hat{j}\phi}
		\end{equation*}
		
\end{section}

\begin{section}{Conversion}
	
	\subsection{Coordinate to Phasor}
		\begin{equation*}
			\begin{cases}
				x = A\cos{\phi}\\
				y = A\sin{\phi}
			\end{cases}
		\end{equation*}
	
	\subsection{Phasor to Coordinate}
		\begin{equation*}
			\begin{cases}
				A = \sqrt{x^2+y^2}\\
				\phi = \arctan(y/x)
			\end{cases}
		\end{equation*}
	
\end{section}
